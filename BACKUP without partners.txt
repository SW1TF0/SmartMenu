        let currentLang = localStorage.getItem("preferredLang") || "bg";
        const langToggleBtn = document.getElementById("langToggle");
        const langTextSpan = document.getElementById("langText");

        function setLanguage(lang) {
            const elements = document.querySelectorAll("[data-i18n]");
            elements.forEach(el => {
                const key = el.getAttribute("data-i18n");
                if (translations[lang][key]) { el.innerHTML = translations[lang][key]; }
            });
            if (lang === "bg") {
                langTextSpan.textContent = "EN"; document.documentElement.lang = "bg";
            } else {
                langTextSpan.textContent = "BG"; document.documentElement.lang = "en";
            }
            localStorage.setItem("preferredLang", lang); currentLang = lang;
        }

        langToggleBtn.addEventListener("click", () => {
            const newLang = currentLang === "bg" ? "en" : "bg";
            setLanguage(newLang);
            if(window.innerWidth <= 768) {
                navLinksMenu.classList.remove('active');
                menuIcon.classList.remove('fa-times');
                menuIcon.classList.add('fa-bars');
            }
        });

        setLanguage(currentLang);
